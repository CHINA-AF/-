    function IsPassFileExt(arrExt, fileExt) {
        var IsFalse = false;
        var ext = arrExt.split(',');
        for (var i = 0; i < ext.length; i++) {
            if (ext[i].toLowerCase() == fileExt.toLowerCase()) {
                IsFalse = true;
            }
        }
        return IsFalse;
    }

    //判断文件大小
    //1 合格，-1未空文件，-2 超过maxSize -3无法判断什么浏览器 -4ie中获取 文件大小失败
    function FileSize(target, maxSize) {
        var browserCfg = {};
        var ua = window.navigator.userAgent;
        if (ua.indexOf("MSIE") >= 1)
            browserCfg.ie = true;
        else if (ua.indexOf("Firefox") >= 0)
            browserCfg.firefox = true;
        else if (ua.indexOf("Chrome") >= 0)
            browserCfg.chrome = true;
        else if (ua.indexOf("Mozilla") >= 0)
            browserCfg.firefox = true;

        var fileSize;
        if (browserCfg.firefox || browserCfg.chrome)
            fileSize = target.files[0].size;
        else if (browserCfg.ie) {
            try {
                var filePath = target.value;
                var fileSystem = new ActiveXObject("Scripting.FileSystemObject");
                var file = fileSystem.GetFile(filePath);
                fileSize = file.Size;
            }
            catch (ex) {
                return -4;
            }
        }
        else
            return -3;

        if (fileSize == 0)
            return -1;
        else if (fileSize / 1024 / 1024 > maxSize)
            return -2;
        else
            return 1;
    }
